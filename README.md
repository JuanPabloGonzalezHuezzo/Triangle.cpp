using namespace std;
  void triangle(int n) {
      for (int a=1; a<=n; a++){
        for (int b=1; b<=a; b++){
          cout << "t";
        }
        cout << endl;
      }
      for (int c=n-1; c>=0; c--){
          for (int d=1; d<=c; d++){
              cout << "t";
          }
          cout << endl;
      }
      // should we use "return;" in here?
  }

  int x;

  cin >> x;
